      const filteredProducts = products.filter(p => {

        const matchCat = activeCategory === 'Semua' || p.category === activeCategory;

        const matchSearch = p.name.toLowerCase().includes(searchQuery.toLowerCase());

        return matchCat && matchSearch;

      });


      const addToCart = (product) => {

        setCart(prev => {

          const existing = prev.find(item => item.id === product.id);

          if (existing) {

            return prev.map(item => item.id === product.id ? { ...item, qty: item.qty + 1 } : item);

          }

          return [...prev, { ...product, qty: 1 }];

        });

      };


      const updateQty = (id, delta) => {

        setCart(prev => prev.map(item => {

          if (item.id === id) {

            const newQty = item.qty + delta;

            return newQty > 0 ? { ...item, qty: newQty } : item;

          }

          return item;

        }));

      };


      const removeFromCart = (id) => setCart(prev => prev.filter(item => item.id !== id));

      const totalBelanja = cart.reduce((sum, item) => sum + (item.price * item.qty), 0);


      const handleCheckout = () => {

        if (cart.length === 0) return;

        setShowPayment(true);

        setPaymentAmount('');

      };

        const bayar = parseInt(paymentAmount.replace(/\D/g, '')) || 0;

        if (bayar < totalBelanja) {

          showAlert({ icon: 'error', title: 'Uang Kurang' });

          return;

        }

          id: invoiceId, 

          date: new Date().toISOString(), 

          cashierId: cashierId, 

          items: cart, 

          total: totalBelanja, 

          payment: bayar, 

          change: bayar - totalBelanja,

          method: paymentMethod // <--- Kirim metode yang dipilih

        };

        

        finishTransaction(transactionData);

      };


      const finishTransaction = (transactionData) => {

        setLastInvoice(transactionData);

        onTransactionSuccess(transactionData);

        setCart([]);

        setShowPayment(false);

        setIsMobileCartOpen(false); // Tutup keranjang di HP setelah berhasil bayar

    function AdminHistory({ transactions, storeConfig }) {


      // LOGIKA UNTUK MENGHASILKAN PDF

      const handleDownloadPDF = () => {

        if(transactions.length === 0) {

           showAlert({ icon: 'info', title: 'Kosong', text: 'Belum ada transaksi untuk didownload.' });

           return;

        }


        try {

          const { jsPDF } = window.jspdf;

          const doc = new jsPDF();


          // HEADER DOCUMENT PDF

          doc.setFontSize(18);

          doc.text(`Laporan Rekap Transaksi`, 14, 22);

          

          doc.setFontSize(12);

          doc.text(`Toko: ${storeConfig.name}`, 14, 30);

          

          doc.setFontSize(10);

          doc.text(`Dicetak pada: ${new Date().toLocaleString('id-ID')}`, 14, 36);


          // FORMAT KOLOM DAN BARIS UNTUK AUTOTABLE

          const tableColumn = ["No", "ID Transaksi", "Kasir", "Tanggal", "Rincian Item", "Total (Rp)"];

          const tableRows = [];

          let grandTotal = 0;


          transactions.forEach((trx, index) => {

            const trxDate = new Date(trx.date).toLocaleString('id-ID');

            const itemsText = trx.items.map(i => `${i.qty}x ${i.name}`).join("\n");

            grandTotal += trx.total;


            const trxData = [

              (index + 1).toString(),

              trx.id,

              trx.cashierId || '-',

              trxDate,

              itemsText,

              trx.total.toLocaleString('id-ID')

            ];

            tableRows.push(trxData);

          });


          // TAMBAHKAN BARIS GRAND TOTAL DI PALING BAWAH

          tableRows.push(["", "", "", "", "GRAND TOTAL KESELURUHAN", grandTotal.toLocaleString('id-ID')]);


          // GENERATE TABEL PDF

          doc.autoTable({

            head: [tableColumn],

            body: tableRows,

            startY: 45,

            theme: 'grid',

            styles: { fontSize: 9, cellPadding: 3 },

            headStyles: { fillColor: [79, 70, 229] }, // Warna Indigo (Sesuai UI)

            columnStyles: {

              0: { cellWidth: 10 },

              4: { cellWidth: 60 } // Kolom rincian item dibuat lebih lebar

            },

            willDrawCell: function (data) {

               // Membuat baris grand total menjadi Bold dan berwarna abu-abu

               if (data.row.index === tableRows.length - 1) {

                   doc.setFont("helvetica", "bold");

                   data.cell.styles.fillColor = [240, 240, 240];

               }

            }

          });


          // SAVE FILE

          doc.save(`Rekap_Transaksi_${new Date().getTime()}.pdf`);

          showAlert({ icon: 'success', title: 'Berhasil', text: 'File PDF berhasil diunduh!' });


        } catch (error) {

          console.error(error);

          showAlert({ icon: 'error', title: 'Gagal', text: 'Terjadi kesalahan saat memproses PDF.' });

        }

      };


      return (

        <div className="bg-white rounded-2xl shadow-sm border border-gray-100">

          {/* HEADER MENU RIWAYAT DENGAN TOMBOL PDF */}

          <div className="p-6 border-b border-gray-100 flex flex-col md:flex-row md:justify-between md:items-center gap-4">

            <h3 className="font-bold text-lg text-gray-800 flex items-center gap-2">

              <Icon name="history" className="text-indigo-600"/> Riwayat Lengkap & Rekap

            </h3>

            

            {/* TOMBOL DOWNLOAD PDF */}
